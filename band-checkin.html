Google Apps Script (Write-Back)

Setup — same script as volunteer page

  1. Open your band roster Google Sheet → Extensions → Apps Script
  2. Paste the script below, save, and deploy as a Web App (Execute as: Me, Anyone can access)
  3. Copy the deployment URL and paste above
function doPost(e) { try { var d = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.openById(d.sheetId); var sheet = ss.getSheets()[d.sheetIndex || 0]; var col = d.col.toUpperCase().charCodeAt(0) - 64; var row = parseInt(d.row); if (d.value === '' || d.value === null) { sheet.getRange(row, col).clearContent(); } else { sheet.getRange(row, col).setValue(d.value); } return ContentService .createTextOutput(JSON.stringify({ok: true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false,error:String(err)})) .setMimeType(ContentService.MimeType.JSON); } } function doGet() { return ContentService.createTextOutput('MPF Band Check-In — active'); }
